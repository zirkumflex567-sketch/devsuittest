## üåê Runtime Language Switching

```csharp
public class LanguageManager : MonoBehaviour
{
    public void SetLanguage(string languageCode)
    {
        // languageCode: "en" or "de"
        LocalizationSettings.SelectedLocale = 
            LocalizationSettings.AvailableLocales.GetLocale(new Locale(languageCode));
        
        PlayerPrefs.SetString("SelectedLanguage", languageCode);
        
        // Refresh all UI texts
        LocalizationManager.RefreshAllUI();
    }
}
```
